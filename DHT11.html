Ինչպես միացնել DHT11 կամ DHT12

DHT11-ը թվային սենսոր է, որը բաղկացած է թերմիստորից և խոնավության սենսորից: DHT11-ն ունի հետևյալ բնութագրերը՝ ստանում է 3,5-5 Վ հոսանք, ջերմաստիճանի որոշում՝ 0-ից 50 աստիճան, 2 աստիճան ճշգրտությամբ, խոնավության որոշում՝ 20%-ից 95%, 5% ճշգրտությամբ։

DHT11 մոդուլը բաղկացած է երեք պիներից և միացվում է հետևյալ սխեմային համաձայն. 

        G - Միացրեք GND
        V - Միացրեք +5V 
        S - Միացեք թվային պինի ( օրինակ, Pin2)
Թերմիստորը ջերմային դիմադրություն է, որի դիմադրությունը փոխվում է ջերմաստիճանից կախված, այսինքն՝ ջերմաստիճանի բարձրացումը հանգեցնում է նրա դիմադրության անկմանը: Հիմնականում թերմիստորը դիմադրության ջերմաչափ է, որը պատրաստված է մետաղների խառը օքսիդներից: Այն կարող է օգտագործվել տարբեր միջավայրերում ջերմաստիճանի ավտոմատ չափման համար: Խոնավություն չափող սենսորը կոնդենսատոր է, որը պարունակում է հաղորդիչներ՝ պղնձե թիթեղներից: Այս կոնդենսատորը տեղադրված է կնքված պատյանում, որի վերևում կա խոնավություն կլանող շերտ: Երբ ջրի մասնիկները հարվածում են այս շերտին, նրա դիէլեկտրական հաստատունը փոխվում է, ինչը հանգեցնում է կոնդենսատորի հզորության փոփոխության:

Անհրաժեշտ էլեմենտներ

  • Arduino Uno / Arduino Nano / Arduino Mega;
  • DHT11 / DHT12 սենսոր
  • breadboard
  • 2 լեդ և 2 հատ 220 Օմ դիմադրություն
  • լարեր «պապա-պապա», «պապա-մամա»

DHT11 սենսորի հետ աշխատելու համար անհրաժեշտ է գրադարանը տեղադրել : Դա անելու համար դուք պետք է ներբեռնեք արխիվը հղումից , DHT11 սենսորն օգտագործելիս պետք է գրադարանը կոդի մեջ ներմուծել : Ներբեռնեք հետևյալ ծրագիրը՝ DHT11 օդի ջերմաստիճանի սենսորը Arduino-ին միացնելուց հետո: 

        #include       // ներմուծում ենք գրադարանը
            DHT dht(2, DHT11);  // հաղորդում, թե որ պորտին է միացված
            
             void setup() {
               dht.begin();                // թողարկում ենք DHT11 սենսորը
               Serial.begin(9600);   
            }
            
            void loop() {
               // հաշվում ենք ջերմաստիճանը  (t) и խոնավությունը(h)
               float h = dht.readHumidity();
               float t = dht.readTemperature();
            
               // արտածում ենք ջերմաստիճանը (t) и խոնավությունը (h) էկրանին
               Serial.print("Խոնավություն՝  ");
               Serial.println(h);
               Serial.print("Ջերմաստիճան՝  ");
               Serial.println(t);
            }

Կոդի մեկնաբանություն

  1. «h» և «t» փոփոխականները float տիպի են, որն օգտագործվում է լողացող կետով թվերը պահելու համար (Arduino-ին միանալիս)
  2. Serial.print() հրամանը տեղեկություն է ուղարկում պորտին առանց տողերի ընդհատման, իսկ Serial.println()հրամանը տեղեկություն է հաղորդում պորտին նոր տողից:

DHT11 և LCD 1602 միացումը Arduino-ին

        #include                        // գրադարան I2C-ի համար 
            #include  //ներմուծում ենք գրադարանը LCD 1602-ի համար
            #include                          // Գրադարան DHT11 համար
            
            LiquidCrystal_I2C LCD(0x27,16,2);  // LCD-ին անուն ենք վերագրում
            DHT dht(2, DHT11);                          // սենսորի միացումը պորտին
            
             void setup() {
               LCD.init();            //  LCD ինիցիալիզացիա
               LCD.backlight();  // էկրանի լույսի միացում
               dht.begin();         // թողարկում ենք  DHT11 սենսորը
            }
            
            void loop() {
              
               float h = dht.readHumidity();
               float t = dht.readTemperature();
            
               // ջերմաստիճանը (t) և խոնավությունը (h) արտածում ենք LCD վրա
               LCD.setCursor(0,0);
               LCD.print("Խոնավություն՝  ");
               LCD.print(h);
            
               LCD.setCursor(0,1);
               LCD.print("Ջերմաստիճան՝  ");
               LCD.print(t);
            
               delay(1000);
               LCD.clear();
            }

Կոդի մեկնաբանություն

  • Սենսորը միկրոկոնտրոլերին միացնելու համար կարող եք օգտագործել ցանկացած ընդհանուր նշանակության պին
  • Ծրագրի LCD.clear()հրամանը մաքրում է ցուցադրման էկրանը գրվածքից՝ DHT11 Arduino ջերմաստիճանի սենսորի նոր արժեքներ ցուցադրելու համար:
  • Եթե անհրաժեշտ է ջերմաստիճանի և խոնավության արժեքներ ստանալ առանց լողացող կետով թվերի, float տիպի փոխարեն օգտագործեք int տիպը
