Ardunio-ի յոթսեգմենտային ցուցիչ

Մոդուլը փոքր լուսադիոդային ցուցիչ է, որում կան յոթ լեդ (դրա շնորհիվ ցուցիչը ստացել է իր անունը) և ութերորդ լեդ-ը՝ կետի տեսքով: Միացնելով լեդերը Arduino Uno- ից տարբեր հաջորդականությամբ ՝ կարող եք տարբեր թվեր ցուցադրել
Պետք է նկատի ունենալ, որ վահանակը չունի ռեզիստորներ, ուստի լեդերը միացնելիս օգտագործեք արտաքին դիմադրություններ:

Անհրաժեշտ էլմենտները

  • Arduino Uno/Ardunio Nano/Ardunio Mega
  • Յոթ սեգմենտային ցուցիչ 5161as/hdsp7503
  • Տակտային սեղմակ
  • ռեզիստորներ 220օմ
  • breadboard
  • Լարեր «պապա-պապա»

Այս օրինակում մենք հերթով կմիացնենք լեդեր ՝ վահանակի վրա տարբեր թվեր նշելու համար: 

 #define  A 8
 # define B 7
 # define B 7
 # define C 6
 # define D 5
 # define E 4
 # define F 3
 # define G 2

void  setup () {
   pinMode (A, OUTPUT  );
   pinMode (B, OUTPUT );
   pinMode (C, OUTPUT  );
   pinMode (D, OUTPUT  );
   pinMode (E, OUTPUT  );
   pinMode (F, OUTPUT );
   pinMode (G, OUTPUT);
}

 void  loop () {
 digitalWrite (A, LOW); //թիվ 1 
 digitalWrite (B, HIGH);
 digitalWrite (C, HIGH);
 digitalWrite (D, LOW);
 digitalWrite (E, LOW);
 digitalWrite (F, LOW);
 digitalWrite (G, LOW);
 delay (1000);
 digitalWrite (A, HIGH); //թիվ 2 
 digitalWrite (B, HIGH);
 digitalWrite (C, LOW);
 digitalWrite (D, HIGH);
 digitalWrite (E, HIGH);
 digitalWrite (F, LOW); 
 digitalWrite (G, LOW);
 delay (1000);
 digitalWrite (A,HIGH); //թիվ 3 
 digitalWrite (B,HIGH);
 digitalWrite (C,HIGH);
 digitalWrite (D,HIGH);
 digitalWrite (E,LOW);
 digitalWrite (F,LOW );
 digitalWrite (G,HIGH);
 delay (1000);
}
