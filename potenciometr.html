Պոտենցիոմետր աշխատանքի սկզբունքը

Պոտենցիոմետր օգնությամբ էլեկտրական շղթայում փոթողվում է դիմադրությունը 0-ից մինչ 10կօմ:
Atmega միկրոկոնտրոլերը պարունակում է բանալոգային փոխակերպիչ (ԱԹՓ), այն 10 բիթ էինչը թույլ է տալիս ստանալ 0-1023 միջակայքից ցանկացախ արժեք: Սանդխակը փոքր արժեքների բաժանելը թույլ է տալիս մեծ ճշգրտությամբ ստանալ ցանկացած ֆիզիկական մեծության արժեք:Որպեսզի հաշվենք անալոգային մութքերից ստացած արժեքը կիռարվում էլեկտրական analogRead ֆունկցիան:

Անրաժեշտ էլեմնտներ

  • Ardunio Uno/Ardunio Nano/Ardunio/Mega
  • breadboard
  • Պոտենցիոմետր
  • 1 լեդ լույս, ռեզիստոր 220օմ
  • լարեր <<Պապա-պապա>>

Պոտենցիոմետրի օգնությամբ կարող ենք սահուն փոփոխել լեդ լույսի պայծառությունը, միացնելով 9-րդ պինին: 


                #define LED_PIN 9 //փոփոխականի սահմանում լեդի համար
                #define POT_PIN A0  //փոփոխականի սահմանում պոտենցիոմետրի համար
            
                void setup()
                {
              
                pinMode(LED_PIN, OUTPUT);
                pinMode(POT_PIN, INPUT);
                Serial.begin(9600);
                }
            
                void loop()
                {
              
                int rotation, brightness;  
            
             
                rotation = analogRead(POT_PIN);
                brightness = rotation / 4;
            
               
                analogWrite(LED_PIN, brightness);
              
                Serial.println(rotation);
                delay(1000);
            }

Կոդի մեկնաբանություն

1. #define օգնությամբ ցույց ենք տալիս, որ 9-րդ պինի անունը LED-PIN է,իսկ անալոգային ելք AO` POT-PIN
2. լոկալ փոփոխական rotation ստանում է 0-1023 միջակայքից թվային արժեքներ
3. brightness փոփոխականը հավասար է rotation/4 և գտնվում է 0-255 միջակայքից: 

                    #include < Servo.h > // գրադարանի ներմուծում
                    Servo servo;            // "servo" տիպի փոփոխականի հայտարարում
                        
                    void setup() {
                        servo.attach(10);        // սերվոն միացվում է պին 10-ին
                        pinMode(A1, INPUT); // A1-ին միացվում է պոտենցիոմետրը 
                    }
                        
                    void loop() {
                        int val = analogRead,/(A1);         
                        val = map(val, 0, 1023, 0, 180); //  val նոր արժեք է հաղորդվում նշված միջակայքից
                        servo.write(val);                  
                        }

Կոդի մեկնաբանություն

mapֆունկցիայի օգնությամբ val փոփոխականի արժեքը 0-1023միջակայքից նոր միջակայք է վերցնում ՝ 0-180:
